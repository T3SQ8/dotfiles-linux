	set border 3

	set yrange [-20:110]

	set ytics add (0 0)
	unset xtics
	set ytics nomirror

	set ylabel '$T$ ($C^{\circ}$)' rotate by 0
	set label 'Smältvärme' at 1,-5
	set label 'Ångbildningsvärme' at 7,95

	plot 0 dashtype 2 lw 2, 100 dashtype 2 lw 2, '-' with lines dashtype 1 lw 2
	0	-10
	1	0
	3	0
	7	100
	10	100
